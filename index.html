const endpoint = 'https://ipapi.co/json/'; fetch(endpoint) .then(response => response.json()) .then(ipData => { const ipAddress = ipData.ip; const geoEndpoint = `http://ip-api.com/json/${ipAddress}?fields=status,message,country,countryCode,region,regionName,city,zip,lat,lon,timezone,isp,org,as`; fetch(geoEndpoint) .then(response => response.json()) .then(data => { if (data.status !== 'success') { document.getElementById('output').innerText = `Error: ${data.message}`; return; } const formattedData = ` IP Address: ${ipAddress} Country: ${data.country} Country Code: ${data.countryCode} Region: ${data.regionName} City: ${data.city} ZIP Code: ${data.zip} Latitude: ${data.lat} Longitude: ${data.lon} Timezone: ${data.timezone} ISP: ${data.isp} Organization: ${data.org} AS: ${data.as} `; document.getElementById('output').innerText = formattedData; }) .catch(error => { document.getElementById('output').innerText = `Error fetching geolocation data: ${error.message || 'An unknown error occurred.'}`; // Provide a more informative message }); }) .catch(error => { document.getElementById('output').innerText = `Error fetching IP address: ${error.message || 'An unknown error occurred.'}`; // Provide a more informative message });
